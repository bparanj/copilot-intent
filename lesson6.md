```javascript
/**
 * @param {string} s
 */
// Write a function that returns the last character of the input string.
function lastCharacter(s) {
  return s[s.length - 1];
}
// Print the result to the console
console.log('lastCharacter("boba"):', lastCharacter("boba"));
```
