```javascript
/**
 * @param {number} x
 * @returns {string}
 */
// Write a function to convert a given input number to a string.
// Input: 123
// Output: "123"
// Input: 123.456
// Output: "123.456"
function convertToString(x) {
  return x.toString();
}
console.log(convertToString(123));
console.log(convertToString(123.456));
```
