```javascript
// Create an arrow function to convert gallons to liters
// 1 gallon = 3.78541 liters
// Display the result to the console
// Call the function
const gallonsToLiters = (gallons) => {
  return gallons * 3.78541;
};
console.log(gallonsToLiters(10));
```
