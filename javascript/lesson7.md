```javascript
/**
 * @param {string} s
 */
// Write a function that returns all the characters except the first one from the input parameter.
// Example
// For s = "Hello!", the output should be
// removeFirstChar(s) = "ello!".
function removeFirstChar(s) {
  return s.slice(1);
}
// log the result
console.log(removeFirstChar("Hello!"));
```
